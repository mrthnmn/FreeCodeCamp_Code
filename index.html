/* This function takes two arrays and returns an array with elements found only in one of the arrays (the symmetric difference of the arrays). */ function diff(arr1, arr2) { var newArr = []; var combinedArr = arr1.concat(arr2).sort(); // Same, same; but different. for (i = 0; i < combinedArr.length; i++); { var iPlusOne = i + 1; // console.log(combinedArr); if (combinedArr[i] === combinedArr[iPlusOne]) { combinedArr.splice([iPlusOne], 1); } else newArr.push(combinedArr[i]); } return newArr; }
